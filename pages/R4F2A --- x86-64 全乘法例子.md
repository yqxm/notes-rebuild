- ## C
	- ``` C
	  	  #include <inttypes.h>
	  	  #include <stdint.h>
	  	  
	  	  typedef unsigned __int128 uint128_t;
	  	  
	  	  void mul(uint128_t *dest, uint64_t x, uint64_t y) {
	  	     *dest = x * (uint128_t)y;
	  	  }
	  ```
- ## ASM
	- ``` asm
	  	  ...
	  	  mul:
	  	  .LFB0:
	  	  	.cfi_startproc
	  	  	endbr64
	  	  	movq	%rsi, %rax			将x放在%rax中
	  	  	mulq	%rdx				无符号全乘法
	  	  	movq	%rax, (%rdi)		复制低64位到内存
	  	  	movq	%rdx, 8(%rdi)		复制高64位到内存
	  	  	ret
	  	  ...
	  ```
	- 这是在小端法机器上的运行结果
